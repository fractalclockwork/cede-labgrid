cmake_minimum_required(VERSION 3.13)

# Default RP2040 Pico; Pico W: -DPICO_BOARD=pico_w (lab/docker Makefile PICO_BOARD=pico_w).
set(PICO_BOARD pico CACHE STRING "Board type")

if(NOT DEFINED ENV{PICO_SDK_PATH})
    message(FATAL_ERROR "PICO_SDK_PATH is not set (set by pico-dev container)")
endif()

include($ENV{PICO_SDK_PATH}/pico_sdk_init.cmake)

project(hello_lab C CXX ASM)

pico_sdk_init()

# Embed build identity in the USB banner (git short hash or CEDE_IMAGE_ID env).
set(_REPO_ROOT "${CMAKE_CURRENT_SOURCE_DIR}/../../..")
if(DEFINED ENV{CEDE_IMAGE_ID} AND NOT "$ENV{CEDE_IMAGE_ID}" STREQUAL "")
  set(CEDE_IMAGE_ID "$ENV{CEDE_IMAGE_ID}")
else()
  execute_process(
    COMMAND git -C "${_REPO_ROOT}" rev-parse --short=12 HEAD
    OUTPUT_VARIABLE CEDE_IMAGE_ID
    OUTPUT_STRIP_TRAILING_WHITESPACE
    ERROR_QUIET
  )
endif()
if(NOT CEDE_IMAGE_ID)
  set(CEDE_IMAGE_ID "unknown")
endif()
string(REGEX REPLACE "[^A-Za-z0-9._-]" "" CEDE_IMAGE_ID "${CEDE_IMAGE_ID}")
if(CEDE_IMAGE_ID STREQUAL "")
  set(CEDE_IMAGE_ID "unknown")
endif()
file(
  WRITE "${CMAKE_CURRENT_BINARY_DIR}/cede_build_id.h"
  "/* Generated by CMake — do not edit. */\n"
  "#ifndef CEDE_BUILD_ID_H\n"
  "#define CEDE_BUILD_ID_H\n"
  "#define CEDE_IMAGE_ID \"${CEDE_IMAGE_ID}\"\n"
  "#endif\n"
)

add_executable(hello_lab src/main.c)
target_include_directories(hello_lab PRIVATE "${CMAKE_CURRENT_BINARY_DIR}")
target_link_libraries(hello_lab pico_stdlib hardware_i2c pico_i2c_slave)
pico_enable_stdio_usb(hello_lab 1)

pico_add_extra_outputs(hello_lab)
